import com.sun.net.httpserver.*;
import java.io.*;
import java.net.*;
public class Ass8 {
	public static void main(String[] args) throws Exception {
		HttpServer server = HttpServer.create(new InetSocketAddress(3012), 0);
		server.createContext("/add", (exchange) -> {
			String query = exchange.getRequestURI().getQuery();
			String[] params = query.split("&");
			int a = Integer.parseInt(params[0].split("=")[1]);
			int b = Integer.parseInt(params[1].split("=")[1]);
			int result = a + b;
			String response = "Result = " + result;
			exchange.sendResponseHeaders(200, response.length());
			OutputStream os = exchange.getResponseBody();
			os.write(response.getBytes());
			os.close();
		});
		server.start();
		System.out.println("Server running at http://localhost:8000");
	}
}

package webservice;

import javax.jws.WebMethod;
import javax.jws.WebService;

@WebService
public interface CalculatorService {

    @WebMethod
    int add(int a, int b);

    @WebMethod
    int sub(int a, int b);
}

CalculatorServiceImpl.java

package webservice;

import javax.jws.WebService;

@WebService(endpointInterface = "webservice.CalculatorService")
public class CalculatorServiceImpl implements CalculatorService {

    public int add(int a, int b) {
        return a + b;
    }

    public int sub(int a, int b) {
        return a - b;
    }
}

Server.java

package webservice;

import javax.xml.ws.Endpoint;

public class Server {

    public static void main(String[] args) {

        Endpoint.publish(
                "http://localhost:8080/calculator",
                new CalculatorServiceImpl());

        System.out.println("Web Service Started...");
    }
}

Client.java

package webservice;

import java.net.URL;
import javax.xml.namespace.QName;
import javax.xml.ws.Service;

public class Client {

    public static void main(String[] args) throws Exception {

        URL url = new URL(
                        "http://localhost:8080/calculator?wsdl");

        QName qname =
                new QName(
                        "http://webservice/",
                        "CalculatorServiceImplService");

        Service service =
                Service.create(url, qname);

        CalculatorService calc =
                service.getPort(CalculatorService.class);

        System.out.println("Addition = "
                + calc.add(10, 5));

        System.out.println("Subtraction = "
                + calc.sub(10, 5));
    }
}

Name the project (e.g., MyWebService) and select GlassFish as the server.
Right-click the project → New → Web Service.
Give the Web Service a name (e.g., Calculator) and package name.
NetBeans will generate a @WebService annotated class automatically.
Add web service operations (methods) inside the class, e.g.:
add(int a, int b)

Right-click project → Deploy to deploy the service on GlassFish.
