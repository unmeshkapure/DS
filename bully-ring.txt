
import java.util.*;

public class Bully {
	
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		System.out.println("Enter number of nodes: ");
		int n = sc.nextInt();
		
		int[] process = new int[n];
		boolean[] active = new boolean[n];
		
		System.out.println("Enter process IDs: ");
		for(int i=0; i<n; i++) {
			process[i] = sc.nextInt();
			active[i] = true;
		}
		
		System.out.println("Enter process ID to fail: ");
		int crash = sc.nextInt();
		
		for(int i=0; i<n; i++) {
			if(process[i] == crash)
				active[i] = false;
		}
		
		
		System.out.println("Enter initiator process: ");
		int initiator = sc.nextInt();
		
		int coordinator = initiator;
		
		for(int i=0; i<n; i++) {
			if(process[i] > initiator && active[i]) {
				System.out.println("Election sent from process "+initiator+" to process "
						+ process[i]);
				coordinator = process[i];
			}
		}
		
		if(coordinator == initiator) {
			System.out.println("No higher process found");
		}
		
		System.out.println("\nNew coordinator is process "+coordinator);
		sc.close();
		
	}
}




import java.util.*;

public class Ring {
	
	static int n;
	static int[] pid;
	static boolean[] active;
	
	static void ring(int initiator) {
		System.out.println("Process "+pid[initiator]+" starts election");
		int current = initiator;
		int maxPid = pid[initiator];
		
		do {
			int next  = (current + 1) % n;
			while(!active[next])
				next = (next + 1) % n;
			
			System.out.println("Ring is passed from Process "
					+ " Process "+pid[current]+" to "+pid[next]);
			if(pid[next] > maxPid)
				maxPid = pid[next];
			
			current = next;
			
		}while(current != initiator);
		
		System.out.println("Process "+maxPid+" becomes new initiator");
	}
	

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		System.out.println("Enter number of processes: ");
		n = sc.nextInt();
		
		pid = new int[n];
		active = new boolean[n];
		
		System.out.println("Enter process ID's: ");
		for(int i=0; i<n; i++) {
			pid[i] = sc.nextInt();
			active[i] = true;
		}
		
		System.out.println("Enter process ID to crash: ");
		int crash = sc.nextInt();
		
		for(int i=0; i<n; i++) {
			if(pid[i] == crash)
				active[i] = false;
		}
		
		System.out.println("Enter initiator process ID: ");
		int initiator = sc.nextInt();
		
		for(int i=0; i<n; i++) {
			if(pid[i] == initiator)
				ring(pid[i]);
		}
		
	}
}
